% ── Paleta de colores por tipo espectral ─────────────────────────────────────
\definecolor{colO}{RGB}{30,100,200}

\definecolor{colNew}{RGB}{0,140,130}

\tikzset{
  %% Nodo de decisión
  dec/.style={
    diamond, draw=gray!60, fill=colDec,
    minimum width=3.6cm, minimum height=1.5cm,
    text width=3.2cm, align=center,
    font=\small, inner sep=2pt, aspect=2.2,
    line width=0.8pt
  },
  %% Resultado (hoja del árbol) — color pasado como argumento
  res/.style 2 args={
    rectangle, rounded corners=5pt,
    draw=#1!80!black, fill=#1!30,
    minimum width=2.8cm, minimum height=1.0cm,
    text width=2.7cm, align=center,
    font=\small\bfseries, line width=1pt,
    text=#1!20!black
  },
  %% Entrada / raíz
  root/.style={
    rectangle, rounded corners=6pt,
    draw=gray!70, fill=gray!15,
    minimum width=4.5cm, minimum height=1.0cm,
    text width=4.3cm, align=center,
    font=\normalsize\bfseries, line width=1pt
  },
  %% Flecha principal
  arr/.style={
    -{Stealth[length=5pt,width=4pt]},
    line width=1.2pt, gray!70
  },
  %% Etiqueta Sí / No
  sy/.style={font=\small\bfseries, text=colNew!80!black, inner sep=1.5pt},
  sn/.style={font=\small\bfseries, text=colM!80!black,  inner sep=1.5pt},
}
